#include<bits/stdc++.h>
#define MAXN 10000000
using namespace std;
int pr[MAXN];
void Prime(int num)
{
    int temp;
    for(int i = 2; i <= num; i++)
    {
        temp = 0;
        for(int j = 2; j < sqrt(i); j++)
        {
            if(i % j == 0)
            {
                temp++;
            }
        }
        if(temp <= 2)
        {
            pr[i] = 1;
        }
    }
}
int main()
{
    int num;
    cin>>num;
    
    for(int i = 2; i <= num; i++)
    {
        if(pr[i] == 1)
        {
            cout<<i<<"\n";
        }
    }
    return 0;
}
